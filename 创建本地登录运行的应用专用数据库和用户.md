创建本地登录运行的应用专用数据库和用户

将localhost替换为%既创建允许远程访问的用户


创建数据库

CREATE DATABASE **jdwx\_db** CHARACTER SET utf8mb4 COLLATE utf8mb4\_unicode\_ci;

创建专用用户（名字，密码）

CREATE USER '**jdwx\_lht**'@'*localhost* ' IDENTIFIED BY '**ddac#afMYSQL%3**';

授予必要权限

GRANT SELECT, INSERT, UPDATE, DELETE ON jdwx\_db.\* TO **'jdwx\_lht**'@'*localhost* ';

刷新权限

FLUSH PRIVILEGES;

退出

EXIT;
